\perlnewcommand\includefile[2]{
    my ($name, $title) = @_;
    
    my $out;
    
    $out = "\\section{$title}\n" if $title;
    
    return "$out\\input{src/$name}";
}

\newcommand{\figureRef}[1]{ \ref{fig:#1} }

\perlnewcommand\figureImage[3]{
	my ($name, $caption, $width) = @_;

    $width = '0.5' unless defined $width;

	return sprintf '\
	\begin{figure}[H]
	\includegraphics[width=%s\textwidth]{%s}
	\caption{%s}
	\label{fig:%s}
	\end{figure}', $width, $name, $caption, $name;
}
